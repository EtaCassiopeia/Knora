@import org.knora.webapi.IRI

@**
 * Given the IRI of a resource, gets the regions that point to this resource.
 *
 * @param triplestore the name of the triplestore being used.
 * @param resourceIri the IRI of the resource.
 *@
 @(triplestore: String,
   resourceIri: IRI)

PREFIX knora-base: <http://www.knora.org/ontology/knora-base#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?region (GROUP_CONCAT(DISTINCT(?regionObjectPermissionAssertion); separator=";") AS ?regionObjectPermissionAssertions) ?owner ?project
@* Ensure that inference is not used in this query. *@
@if(triplestore.startsWith("graphdb")) {
    FROM <http://www.ontotext.com/explicit>
}
WHERE {

    BIND(IRI("@resourceIri") as ?resource)

    ?region knora-base:isRegionOf ?resource .

    ?region a knora-base:Region .

    ?permission rdfs:subPropertyOf+ knora-base:hasPermission .
    ?region ?permission ?regionObjectPermissionGroup .
    BIND(CONCAT(STR(?permission), ",", STR(?regionObjectPermissionGroup)) as ?regionObjectPermissionAssertion)

    ?region knora-base:attachedToUser ?owner .

    ?region knora-base:attachedToProject ?project .

}

GROUP BY
    ?region
    ?owner
    ?project
